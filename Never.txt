-- // NeverLose Again Key System GUI // --

-- Создаём GUI
local ScreenGui = Instance.new("ScreenGui")
local MainFrame = Instance.new("Frame")
local UICorner = Instance.new("UICorner")
local Title = Instance.new("TextLabel")
local KeyBox = Instance.new("TextBox")
local UICorner2 = Instance.new("UICorner")
local Submit = Instance.new("TextButton")
local UICorner3 = Instance.new("UICorner")

-- Добавляем в игрока
ScreenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Настройки основного окна
MainFrame.Parent = ScreenGui
MainFrame.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
MainFrame.BorderSizePixel = 0
MainFrame.Position = UDim2.new(0.35, 0, 0.35, 0)
MainFrame.Size = UDim2.new(0, 400, 0, 220)
MainFrame.BackgroundTransparency = 0.05

-- Закругления
UICorner.CornerRadius = UDim.new(0, 20)
UICorner.Parent = MainFrame

-- Заголовок
Title.Parent = MainFrame
Title.BackgroundTransparency = 1
Title.Size = UDim2.new(1, 0, 0, 60)
Title.Font = Enum.Font.GothamBold
Title.Text = "NeverLose Again - Key System"
Title.TextColor3 = Color3.fromRGB(255, 255, 255)
Title.TextSize = 22
Title.TextStrokeTransparency = 0.9

-- Поле ввода ключа
KeyBox.Parent = MainFrame
KeyBox.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
KeyBox.BorderSizePixel = 0
KeyBox.Position = UDim2.new(0.1, 0, 0.45, 0)
KeyBox.Size = UDim2.new(0, 320, 0, 40)
KeyBox.Font = Enum.Font.Gotham
KeyBox.PlaceholderText = "Enter your key..."
KeyBox.Text = ""
KeyBox.TextColor3 = Color3.fromRGB(255, 255, 255)
KeyBox.TextSize = 16
UICorner2.CornerRadius = UDim.new(0, 10)
UICorner2.Parent = KeyBox

-- Кнопка подтверждения
Submit.Parent = MainFrame
Submit.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
Submit.Position = UDim2.new(0.3, 0, 0.7, 0)
Submit.Size = UDim2.new(0, 160, 0, 40)
Submit.Font = Enum.Font.GothamBold
Submit.Text = "VERIFY KEY"
Submit.TextColor3 = Color3.fromRGB(255, 255, 255)
Submit.TextSize = 18
UICorner3.CornerRadius = UDim.new(0, 10)
UICorner3.Parent = Submit

-- Эффект плавного появления
MainFrame.BackgroundTransparency = 1
Title.TextTransparency = 1
KeyBox.TextTransparency = 1
Submit.TextTransparency = 1

for i = 1, 10 do
	task.wait(0.05)
	MainFrame.BackgroundTransparency = MainFrame.BackgroundTransparency - 0.1
	Title.TextTransparency = Title.TextTransparency - 0.1
	KeyBox.TextTransparency = KeyBox.TextTransparency - 0.1
	Submit.TextTransparency = Submit.TextTransparency - 0.1
end

-- Проверка ключа
local correctKey = game:HttpGet("https://github.com/NeverLoseAgain/Never/blob/main/Key.txt")

Submit.MouseButton1Click:Connect(function()
	if KeyBox.Text == correctKey then
		Submit.Text = "ACCESS GRANTED"
		wait(1)
		ScreenGui:Destroy()
		-- Здесь вставь код запуска своего скрипта ↓
		loadstring(game:HttpGet("https://github.com/NeverLoseAgain/Never/blob/main/NeverLoseScript.txt"))()
	else
		Submit.Text = "INVALID KEY"
		wait(1)
		Submit.Text = "VERIFY KEY"
	end
end)
